## Install

```bash
npm install --save react-materialize-forms
```

## Usage

```jsx
import React, { Component } from 'react'

import { AutoComplete } from 'react-materialize-forms'

class Example extends Component {

  

  render () {
    return (
      <AutoComplete />
    )
  }
}
```
